Webhook Monitor

A real-time dashboard for monitoring anchor webhook callbacks. Displays incoming events with filtering, statistics, and detailed payload inspection.

üíª Usage Example

// Rust usage - Webhook handling
use anchor_kit::webhook::{WebhookMiddleware, WebhookEvent};

let middleware = WebhookMiddleware::new(
  "https://your-server.com/webhook",
  "your-secret-key"
);

// Process incoming webhook
fn handle_webhook(payload: &[u8], signature: &str) {
  if middleware.verify(signature, payload) {
    let event: WebhookEvent = serde_json::from_slice(payload).unwrap();
    // Process event
  }
}
